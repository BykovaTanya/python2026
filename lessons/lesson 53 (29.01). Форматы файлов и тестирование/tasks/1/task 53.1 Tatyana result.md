import csv

def calculate_average_grades(filename='students.csv'):
    with open(filename, 'r', encoding='utf-8') as file:
        reader = csv.DictReader(file)  # Используем DictReader для удобства
        subjects = reader.fieldnames[1:]  # Получаем названия предметов (исключая 'Name')
        totals = {subject: 0 for subject in subjects}
        count = 0
        
        for row in reader:
            for subject in subjects:
                totals[subject] += int(row[subject])
            count += 1
        
        averages = {subject: totals[subject] / count for subject in subjects}
        
        for subject, average in averages.items():
            print(f"{subject}: {average:.1f}")

if __name__ == "__main__":
    calculate_average_grades()
